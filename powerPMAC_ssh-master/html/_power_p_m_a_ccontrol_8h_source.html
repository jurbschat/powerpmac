200  /**
201  * @brief Get variable value.
202  *
203  * This is a template method: the type of 'value' can be float, double, int, unsigned int or std::string.
204  *
205  * If 'inf' or 'nan' is received from the Power PMAC,
206  * this function will return PPMACcontrolPMACUnexpectedReplyError (-231) and
207  * the value parameter will not be set. \n
208  * Power PMAC command string sent = "<name>"
209  *
210  * @param name - Variable name
211  * @param value - Value of the variable - reference to float, double, int or std::string
212  * @return If variable is successfully received,
213  * PPMACcontrolNoError(0) is returned. If not,
214  * minus value is returned. Possible error codes are :
215  * - PPMACcontrolNoError(0)
216  * - Error reported from Power PMAC (-1 to -99) -1*(Power PMAC error number)
217  * - PPMACcontrolNoSSHDriverSet (-230)
218  * - PPMACcontrolSSHDriverError (-102)
219  * - PPMACcontrolSSHDriverErrorNoconn (-104)
220  * - PPMACcontrolSSHDriverErrorNobytes (-103)
221  * - PPMACcontrolSSHDriverErrorReadTimeout (-112)
222  * - PPMACcontrolSSHDriverErrorWriteTimeout (-113)
223  * - PPMACcontrolPMACUnexpectedReplyError (-231)
224  * - PPMACcontrolUnexpectedParamError (-238)
225  * - PPMACcontrolSemaphoreTimeoutError = (-239)
226  * - PPMACcontrolSemaphoreError = (-240)
227  * - PPMACcontrolSemaphoreReleaseError = (-241)
228  *
229  */
230  template <typename T> int PowerPMACcontrol_getVariable(const std::string name, T& value){
231  static const char *functionName = "PowerPMACcontrol::PowerPMACcontrol_getVariable";
232  debugPrint_ppmaccomm("%s called ", functionName);
233 
234  // Check variable name is valid
235  if (name.length() < 1)
236  return PPMACcontrolUnexpectedParamError;
237 
238  // Build the buffer to send
239  char cmd[SEND_BUFFER_LENGTH] = "";
240  this->buildSendBuffer(cmd, name);
241 
242  // Send command and read reply
243  std::string reply;
244  int ret = this->writeRead(cmd,reply);
245  if (ret != PPMACcontrolNoError)
246  return ret;
247 
248  // Convert result to desired type
249  std::istringstream stream(reply);
250  T rval;
251  stream >> rval;
252  if (stream.fail())
253  {
254  // Failed to convert to desired type -
255  // Returns this error when the stream is empty
256  // or contains a reply not in the expected format
257  return PPMACcontrolPMACUnexpectedReplyError;
258  }
259  value = rval;
260  return PPMACcontrolNoError;
261  };
