112 /// If noValue == false, scan for a matching value and return it.
113 /// If noValue == true, don't scan for a value but return 0 to say that the keyword is present
114 int argParser::checkForArg(std::string argName, std::string & value, bool noValue)
115 {
116  // Loop through the supplied command-line arguments.
117  // Look for pairs '-argName value';
118  // or, if noValue == true, look for '-argName' only.
119 
120  int argc = myArgc;
121  char ** argv;
122  argv = myArgv;
123  if (argc > 1)
124  {
125  for (int i = 0; i < argc; i ++)
126  {
127  std::string str(argv[i]);
128  if (str == argName)
129  {
130  // Found the argument keyword
131  if (noValue == true)
132  {
133  // If noValue is set to true then we only check for the presence of the argname
134  // and don't look for a value.
135  return 0;
136  }
137  else
138  {
139  if (i >= (argc - 1))
140  {
141  // There are no more arguments so we can't find a value
142  return -1;
143  }
144  else
145  {
146  // Look for a value in the next argument
147  std::string nextstr(argv[i+1]);
148  if (nextstr.find("-") == 0)
149  {
150  // The next argument is also a keyword (so no value)
151  return -1;
152  }
153  else
154  {
155  // Return a value for this keyword
156  value = argv[i+1];
157  return 0;
158  }
159  }
160  }
161  }
162  } // End for loop
163  }
164 
165  // Gone through all the arguments without finding this one
166  return -1;
